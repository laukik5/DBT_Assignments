1) Create a view that shows all of the customers who have the highest ratings.
Answer:	create view v_customers as select Cname
 	from customers 
        where rating = (select max(rating) from  customers);

2) Create a view that shows the number of salespeople in each city.
Answer:	create view v_salespeople as (select city, count(snum) no_sales from salespeople group by city);
	select * from v_salespeople;


3) Create a view that shows the average and total orders for each salesperson after his or her name. Assume all names are unique.
Answer:	create view v_salescount as  (select Sname,avg(amt) Average, count(Onum) from orders Natural join salespeople group by Snum);
	select * from  v_salescount; 


4) Create a view that shows each salesperson with multiple customers.
Answer:	create view v_mulsalcust as (select salespeople.sname,cname from  customers, salespeople);
select *  from v_mulsalcust;
